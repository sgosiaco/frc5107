006package org.usfirst.frc5107.subsystems;
007
008import edu.wpi.first.wpilibj.Compressor;
009import edu.wpi.first.wpilibj.Solenoid;
010import edu.wpi.first.wpilibj.command.Subsystem;
011import org.usfirst.frc5107.RobotMap;
012import org.usfirst.frc5107.commands.Compressor1;
013
014/**
015 *
016 * @author BITKRUSHER
017 */
018public class Pneumatics extends Subsystem {
019
020    Solenoid solenoid1;
021    Solenoid solenoid2;
022    Compressor comp1;
023    
024    public Pneumatics() {
025        solenoid1 = new Solenoid(3);
026        solenoid2 = new Solenoid(RobotMap.solenoid2);
027        comp1 = new Compressor(1,1);
028    }
029
030    public void initDefaultCommand() {
031        setDefaultCommand(new Compressor1());
032    }
033
034    public void neutral() {
035        solenoid1.set(false);
036        solenoid2.set(true);
037    }
038
039    public void fire() {
040            solenoid1.set(true);
041            solenoid2.set(false);
042    }
043    public void compOn(){
044        comp1.start();
045    }
046    public void compOff(){
047        comp1.stop();
048    }
049    public void compEnable(){
050        comp1.enabled();
051    }
052    public boolean compSwitch(){
053        return comp1.getPressureSwitchValue();
054    }
055
056}
