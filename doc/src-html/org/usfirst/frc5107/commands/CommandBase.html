003import edu.wpi.first.wpilibj.NamedSendable;
004import edu.wpi.first.wpilibj.command.Command;
005import edu.wpi.first.wpilibj.smartdashboard.SmartDashboard;
006import org.usfirst.frc5107.OI;
007import org.usfirst.frc5107.subsystems.Camera;
008import org.usfirst.frc5107.subsystems.Claw;
009import org.usfirst.frc5107.subsystems.DriveTrain;
010import org.usfirst.frc5107.subsystems.Pneumatics;
011
012import org.usfirst.frc5107.subsystems.Camera;
013import org.usfirst.frc5107.subsystems.ClawFeed;
014
015/**
016 * The base for all commands. All atomic commands should subclass CommandBase.
017 * CommandBase stores creates and stores each control system. To access a
018 * subsystem elsewhere in your code in your code use CommandBase.exampleSubsystem
019 * @author Author
020 */
021public abstract class CommandBase extends Command {
022
023    public static OI oi;
024    // Create a single static instance of all of your subsystems
025    public static DriveTrain driveTrain = new DriveTrain();
026    public static Claw claw = new Claw();
027    public static Pneumatics pneumatics = new Pneumatics();
028    public static Camera camera = new Camera();
029    public static ClawFeed clawFeed = new ClawFeed();
030
031    public static void init() {
032        // This MUST be here. If the OI creates Commands (which it very likely
033        // will), constructing it during the construction of CommandBase (from
034        // which commands extend), subsystems are not guaranteed to be
035        // yet. Thus, their requires() statements may grab null pointers. Bad
036        // news. Don't move it.
037        oi = new OI();
038
039        // Show what command your subsystem is running on the SmartDashboard
040        SmartDashboard.putData(driveTrain);
041        SmartDashboard.putData(claw);
042        SmartDashboard.putData(pneumatics);
043        
044    }
045
046    public CommandBase(String name) {
047        super(name);
048    }
049
050    public CommandBase() {
051        super();
052    }
053}
